on_actions = {
	# ROOT is capitulated country, FROM is winner
	on_capitulation = {
		effect = {
			log = "[GetDateText] [Root.GetName] has capitulated to [From.GetName]"
			news_event = country_capitulated.0
			if = {
				limit = {
					tag = FIN
					has_war_with = RFN
				}
				RFN = {
					annex_country = {
						target = FIN
						transfer_troops = no
					}
				}
				RUS = {
				country_event = rus.34

}
			}
			if = {
				limit = {
					AND = {
						ROOT = { tag = SAH }
						FROM = { tag = ALG }
					}
				}
				FRA = {
				country_event = africa.5
			}
			}			
	}
}
}
