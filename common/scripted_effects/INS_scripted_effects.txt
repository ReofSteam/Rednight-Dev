##Kalimantan's Council of Sultans
KAL_Bulungan_levelup = {
	if = {
		limit = {
			has_country_flag = KAL_bulungan_contentious
		}
		clr_country_flag = KAL_bulungan_contentious
		set_country_flag = KAL_bulungan_disloyal
	}
	else_if = {
		limit = {
			has_country_flag = KAL_bulungan_disloyal
		}
		clr_country_flag = KAL_bulungan_disloyal
		set_country_flag = KAL_bulungan_content
	}
	else_if = {
		limit = {
			has_country_flag = KAL_bulungan_content
		}
		clr_country_flag = KAL_bulungan_content
		set_country_flag = KAL_bulungan_sympathetic
	}
	else_if = {
		limit = {
			has_country_flag = KAL_bulungan_sympathetic
		}
		clr_country_flag = KAL_bulungan_sympathetic
		set_country_flag = KAL_bulungan_loyal
	}
}
