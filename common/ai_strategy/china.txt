			tag = YUN

			tag = XSM
			tag = SIK
			tag = PRC
			tag = TAN
			tag = ETS
			tag = TIB
			tag = MON
			tag = LEC
		}
	}
	enable = {
		is_chinese_tag = yes
		NOT = {
			any_neighbor_country = { has_war_with = ROOT }

		}

		is_in_faction_with = QIE
		QIE = {
			exists = yes
			is_faction_leader = yes
			has_capitulated = no
		}
	}

	ai_strategy = {
		type = protect
		id = "QIE"
		value = 200
	}
	ai_strategy = {
		type = garrison
		value = -50
	}
}
