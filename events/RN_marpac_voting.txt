		}

	id = marpac.2
	title = marpac.1.t
	desc = marpac.1
	picture = GFX_FRA_may_pustch
    hidden = yes
	is_triggered_only = yes

	option = {
		name = marpac.1.b
		if = {
		limit = {
		check_variable = {
	    var = marpac_for_vote
	    value = marpac_against_vote
	    compare = greater_than_or_equals
}
	}
	country_event = marpac.3
}	
    else = {
	country_event = marpac.4
	}		
	clr_global_flag = MARPAC_voting_in_process
	clr_country_flag = MARPAC_resolution_51
	clear_array = global.marpac_countries_against
	clear_array = global.marpac_countries_for	

	}	
}
