country_event = {        
	id = marpac.5
	title = marpacnews.t
	desc = marpac.5.d
	picture = GFX_FRA_may_pustch
	is_triggered_only = yes
	trigger = {
	is_ai = yes
	NOT = { has_country_flag = MARPAC_voted }		
	}

	option = {
		name = marpac.5.a	
		ai_chance = { factor = 10 
		modifier = {
		is_in_array = {
    	array = global.marpac_countries_for
      	value = FRA
}
        factor = 7
    }
		}
	set_country_flag = MARPAC_voted	
		    add_to_array = {
	        array = global.marpac_countries_for
            value = THIS.id
		}		
			add_to_variable = {
	        var = global.marpac_for_vote
	        value = 1
        }	
	}	
	option = {
		name = marpac.5.b		
		ai_chance = { factor = 10
		modifier = {
		is_in_array = {
    	array = global.marpac_against_vote
      	value = FRA
}
        factor = 7
    }		
		}		
			set_country_flag = MARPAC_voted	
			add_to_variable = {
	        var = global.marpac_against_vote
	        value = 1
			}
		    add_to_array = {
	        array = global.marpac_countries_against
            value = THIS.id
		}				
	}	
}
